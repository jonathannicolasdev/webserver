server {
  listen 8080;
  listen 1111;
#  listen 1111;
# listen 0.0.0.0:1111;
# listen 127.0.0.1:1111;


  root /www/;
  
	server_name localhost;
	index index.html;

  location / {
    allow_methods GET;
    index index.html;
  }

  location /autoindex {
    autoindex on;
  }

  location /data {
    root /data/;
    index index.html;
    limit_except GET;
    return    https://www.google.com;
  }

  location /my_errors {
    root /errors;
  }

  location /imgs {
    root /assets/;
  }
  location /cgi {
    allow_methods GET POST;
    root /cgi-bin/;
  }

  location /uploads {
    autoindex on;
    allow_methods GET DELETE POST;
    max_body_size 55000
    root /client_uploads/
  }
  
  error_page 404 /default_errors/404.html;
}
