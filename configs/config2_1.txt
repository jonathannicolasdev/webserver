#  listen 1111;

  root /www/;
  
	server_name localhost;
	index index.html;

  location /yangon {
    allow_methods GET;
    index index.html;
  }
  
  location / {
    autoindex on;
    allow_methods GET;
    index index.html;
  }

  location /autoindex {
    autoindex on;
  }

  location /data {
    root /data/;
    index index.html;
    limit_except GET;
    return    https://www.google.com;
  }

  location /my_errors { 
    autoindex on;
    root /errors;
  }

  location /imgs {
    root /assets/;
  }
  location /cgi {
    allow_methods GET POST;
    root /cgi-bin/;
  }
  location /scripts {
    allow_methods GET;
    root /js_scripts/;
  }

  location /uploads {
    autoindex on;
    allow_methods GET DELETE POST;
    max_body_size 10000
    root /client_uploads/
  }
  
  error_page 404 /my_errors/404.html;
}
