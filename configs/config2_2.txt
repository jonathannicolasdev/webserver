server {
  listen 8082;
  listen 65534;

  root /www/;
  
	server_name localhost;
	index index.html;

  location / {
    autoindex on
    allow_methods GET;
    index index.html;
  }

  location /autoindex {
    autoindex on;
  }

  location /data {
    root /data/;
    index index.html;
    limit_except GET;
#    return    https://www.google.com;
    return    localhost:8081/;
  }

  location /errors {
    autoindex on;
    root /my_errors;
  }

  location /imgs {
    root /assets/;
  }
  location /cgi {
    allow_methods GET POST;
    root /cgi-bin/;
  }
  
  location /scripts {
    allow_methods GET;
    root /js_scripts/;
  }

  # Max body size of 100mb to start
  location /uploads {
    root /client_uploads/
    autoindex on;
    allow_methods GET DELETE POST;
    allow_download on;
    allow_delete off;
    max_body_size 100000000
  }
  
  error_page 404 /my_errors2/404.html;
}
